`timescale 1ns / 1ps

module d_ff (d_ff_if vif);
 
  always @(posedge vif.clk)
    begin
      if (vif.reset == 1'b1)
        vif.q <= 1'b0;
      else  vif.q <= vif.d;
    end
  endmodule
 
interface d_ff_if;
  logic clk;   
  logic reset;   
  logic d;   
  logic q;   
endinterface
